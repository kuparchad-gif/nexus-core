# Set cycle length using presets
set_cycle_preset('quick')        # 2 phases, 1 epoch each
set_cycle_preset('standard')     # 4 phases, 2 epochs each  
set_cycle_preset('comprehensive') # 6 phases, 3 epochs each
set_cycle_preset('research')     # 7 phases, 4 epochs each

# Create custom cycles
create_custom_cycle(
    name="my_custom_cycle",
    phases=["phase1", "phase2", "phase3", "phase4"],
    epochs=3,      # Epochs per phase
    samples=75     # Data samples per phase
)
set_cycle_preset("my_custom_cycle")

# Combined turbo + cycle control
set_turbo_mode('turbo')
set_cycle_preset('comprehensive')

# Quick training with both settings
result = quick_train("your_topic", "turbo", "comprehensive")
